set -e

SRC=$( cd "$( dirname "${BASH_SOURCE[0]}" )" && pwd )
SCRIPTNAME=$(basename ${BASH_SOURCE[0]})
VERSION=$(cat "$SRC/VERSION")

help() {
    cat <<EOF
$SCRIPTNAME v$VERSION

Usage:
    $SCRIPTNAME ARG

Parameter:
    ARG         some argument
EOF
}

arg=$1

if [ -z "$arg" ]
then
    help
    exit 1
fi
